docker ps [-a: показывает сохраненные контейнеры] -> показывает все запущенные докер контейнеры
docker rm <идентефикатор контейнера> -> удаляет докер контейнер
docker rmi <идентефикатор image> -> удаляет image
docker run [-d: не забивает терминал, --name: имя контейнерач]<имя image> -> запускает image и создает docker контейнер
docker start <имя контейнера> -> запускает готовый docker контейнер
docker images -> показывает установленные images
docker pull <название image> -> скачивает image c docker hub
docker kill (stop) <контейнер> -> убивает (останавливает) работу контейнера
docker inspect <id контейнера> -> показывает информацию по контейнеру
docker stats <id контейнера> -> показывает, сколько контейнер потребляет ресурсов
docker exec -it <Имя контейнера> /bin/bash -> заходит внутрь контейнера
